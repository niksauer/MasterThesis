    %\def\refmarkup{\labelmarkup}% How to markup the reference

    \else\def\@currentlabel{\refmarkup{#1}}{\label{#3}}\fi%
    \@esphack%
    \labelmarkup{#2}% visible printed text.
}

  \edef\@currentlabel{\number\c@countitems}%
